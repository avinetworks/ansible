Synopsis
--------

Examples
--------

.. code-block:: yaml

    
  - name: Create a string group configuration
    avi_stringgroup:
      controller: '{{ controller }}'
      password: '{{ password }}'
      username: '{{ username }}'
      kv:
      - key: text/html
      - key: text/xml
      - key: text/plain
      - key: text/css
      - key: text/javascript
      - key: application/javascript
      - key: application/x-javascript
      - key: application/xml
      - key: application/pdf
      name: System-Compressible-Content-Types
      tenant_ref: /api/tenant?name=admin
      type: SG_TYPE_STRING
